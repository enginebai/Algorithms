## [841. Keys and Rooms](https://leetcode.com/problems/keys-and-rooms/)

```kotlin
class Solution {
    fun canVisitAllRooms(rooms: List<List<Int>>): Boolean {
        val visited = BooleanArray(rooms.size)
        // dfs(rooms, 0, visited)
        bfs(rooms, visited)
        
        for (i in 0 until rooms.size) {
            if (visited[i] == false) return false
        }
        return true
    }
    
    private fun dfs(rooms: List<List<Int>>, i: Int, visited: BooleanArray) {
        if (visited[i] == true) return
        visited[i] = true
        rooms[i].forEach { room ->
            if (room != i)
                dfs(rooms, room, visited)
        }
    }
    
    private fun bfs(rooms: List<List<Int>>, visited: BooleanArray) {
        val queue = ArrayDeque<Int>()
        queue.addLast(0)
        while (queue.isNotEmpty()) {
            val node = queue.removeFirst()
            visited[node] = true
            rooms[node].forEach { r ->
                if (r != node && visited[r] != true) queue.addLast(r)
            }
        }
    }
}
```

* **Time Complexity**: `O(room + key)`.
* **Space Complexity**: `O(room)`.
