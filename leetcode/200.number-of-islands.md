## [200. Number of Islands](https://leetcode.com/problems/number-of-islands/)

```kotlin
class Solution {
    
    private val visited = hashSetOf<Pair<Int, Int>>()       
    private var numberOfIslands = 0
    
    fun numIslands(grid: Array<CharArray>): Int {
        
        for (m in 0 until grid.size) {
            for (n in 0 until grid[m].size) {
                if (!visited.contains(m to n) && grid[m][n] == '1') {
                    dfs(grid, m, n)
                    numberOfIslands++
                }
            }
        }
        return numberOfIslands
    }
    
    private fun dfs(grid: Array<CharArray>, x: Int, y: Int) {
        if (x < 0 || y < 0 || x >= grid.size || y >= grid[0].size || grid[x][y] == '0' || visited.contains(x to y)) return
        
        visited.add(x to y)
        dfs(grid, x - 1, y)
        dfs(grid, x + 1, y)
        dfs(grid, x, y - 1)
        dfs(grid, x, y + 1)
    }
}
```
