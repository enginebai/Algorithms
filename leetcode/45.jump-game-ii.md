## Dynamic Programming

fun jump(nums: IntArray): Int {
    if (nums.size <= 2) return nums.size - 1
    val dp = IntArray(nums.size) { _ -> Int.MAX_VALUE }
    dp[0] = 0
    dp[1] = 1
    for (i in 2 until nums.size) {
        for (j in i - 1 downTo 0) {
            if (j + nums[j] >= i) {
                dp[i] = min(dp[i], dp[j] + 1)
            }
        }
    }        
    return dp[nums.size - 1]
}
