# [3047. Find the Largest Area of Square Inside Two Rectangles](https://leetcode.com/problems/find-the-largest-area-of-square-inside-two-rectangles/)

## Interval Intersections
```kotlin
data class Rect(
    val bottomLeft: IntArray,
    val topRight: IntArray
)

fun largestSquareArea(bottomLeft: Array<IntArray>, topRight: Array<IntArray>): Long {
    val n = bottomLeft.size
    val rects = Array(n) {
        Rect(bottomLeft[it], topRight[it])
    }
    return byBruteForce(rects)
}

private fun byBruteForce(rects: Array<Rect>): Long {
    val n = rects.size
    rects.sortWith(compareBy<Rect> { it.bottomLeft.first() }.thenBy { it.bottomLeft[1] })
    var maxLength = 0
    for (i in 0 until n) {
        for (j in i + 1 until n) {
            val rect1 = rects[i]
            val rect2 = rects[j]

            val xSide1 = intArrayOf(rect1.bottomLeft[0], rect1.topRight[0])
            val xSide2 = intArrayOf(rect2.bottomLeft[0], rect2.topRight[0])
            val xLength = minOf(xSide1[1], xSide2[1]) - maxOf(xSide1[0], xSide2[0])

            val ySide1 = intArrayOf(rect1.bottomLeft[1], rect1.topRight[1])
            val ySide2 = intArrayOf(rect2.bottomLeft[1], rect2.topRight[1])
            val yLength = minOf(ySide1[1], ySide2[1]) - maxOf(ySide1[0], ySide2[0])

            val squareLength = minOf(xLength, yLength)
            if (squareLength >= 0) {
                maxLength = maxOf(maxLength, squareLength)
            }
        }
    }
    return maxLength.toLong() * maxLength
}

/**
parallel to x and y axis.
   -----* topRight
   |    |
   * ----
bottomLeft

maximum area of a square that can fit inside the intersecting region of at least two rectangles
 */
```
